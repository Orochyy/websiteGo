```mysql
  `id` int(6) unsigned NOT NULL AUTO_INCREMENT,

  PRIMARY KEY (`id`)
```

```mysql
CREATE TABLE users(
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50),
    password VARCHAR(120)
);
```
